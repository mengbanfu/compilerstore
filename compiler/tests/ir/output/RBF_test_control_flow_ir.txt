; ModuleID = 'main'
source_filename = "main"

define i32 @main() {
entry:
  %x = alloca i32, align 4
  store i32 10, ptr %x, align 4
  %y = alloca i32, align 4
  store i32 5, ptr %y, align 4
  %x1 = load i32, ptr %x, align 4
  %y2 = load i32, ptr %y, align 4
  %gt = icmp sgt i32 %x1, %y2
  br i1 %gt, label %then, label %else

then:                                             ; preds = %entry
  %x3 = load i32, ptr %x, align 4
  %y4 = load i32, ptr %y, align 4
  %add = add i32 %x3, %y4
  store i32 %add, ptr %x, align 4
  br label %merge

else:                                             ; preds = %entry
  %x5 = load i32, ptr %x, align 4
  %y6 = load i32, ptr %y, align 4
  %sub = sub i32 %x5, %y6
  store i32 %sub, ptr %x, align 4
  br label %merge

merge:                                            ; preds = %else, %then
  %x7 = load i32, ptr %x, align 4
  ret i32 %x7
}
